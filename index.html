    <div class="title" id="main-text">LOADING CAMERA...</div>

  </div>

  

  <script>

    let video;

    let facemesh;

    let predictions = [];

    

    // Game states

    let gameState = 'LOADING'; 

    let score = 0;

    let isChrome = false;

    

    // Game objects

    let ball = { x: 0, y: 0, vx: 0, vy: 0, radius: 25 };

    let nose = { x: -100, y: -100, radius: 25 };

    let ballParticles = []; 


    function setup() {

      // Chrome Check

      let userAgent = navigator.userAgent;

      let vendor = navigator.vendor;

      isChrome = (/Chrome/.test(userAgent) && /Google Inc/.test(vendor)) || userAgent.match("CriOS");


      if (!isChrome) {

        gameState = 'BLOCKED';

        document.getElementById('main-text').innerText = "CHROME REQUIRED";

        document.getElementById('main-text').style.color = "#ff5555";

        document.getElementById('sub-text').innerText = "OPEN IN GOOGLE CHROME TO PLAY.";

        noCanvas();

        return; 

      }


      createCanvas(windowWidth, windowHeight);

      video = createCapture({ audio: false, video: { facingMode: "user" } });

      video.hide(); 

      

      // Fix for drift, setting canvas to video's raw dimensions

      video.elt.onloadedmetadata = () => {

        resizeCanvas(video.elt.videoWidth, video.elt.videoHeight);

        resetGame();

      };


      facemesh = ml5.facemesh(video, () => {

        document.getElementById('message-container').style.display = 'none';

        gameState = 'ALIGN';

      });

      

      facemesh.on('predict', results => { predictions = results; });

      textAlign(CENTER, CENTER);

    }


    function draw() {

      if (gameState === 'LOADING' || gameState === 'BLOCKED') return;


      background(20);


      // 1. Mirrored video background (No blue filter, natural color)

      push();

      translate(width, 0);

      scale(-1, 1);

      // Removed the tint completely for full natural color

      image(video, 0, 0, width, height);

      pop();


      // Add a very faint dark overlay just so white text remains readable 

      // over bright camera feeds, without shifting the color.

      fill(0, 0, 0, 60);

      rect(0, 0, width, height);


      // 2. Update and draw the Nose Tracker (Subtle)

      if (predictions.length > 0) {

        let n = predictions[0].annotations.noseTip[0];

        nose.x = lerp(nose.x, width - n[0], 0.5); 

        nose.y = lerp(nose.y, n[1], 0.5);

      }

      

      // Draw very faint tracker (subtle feedback)

      noStroke();

      fill(0, 255, 0, 15); 

      ellipse(nose.x, nose.y, nose.radius * 2);


      // --- STATE: ALIGN ---

      if (gameState === 'ALIGN') {

        let targetX = width / 2;

        let targetY = height / 2;

        let d = dist(nose.x, nose.y, targetX, targetY);

        let isAligned = d < 60;


        noFill();

        strokeWeight(4);

        let ringColor = lerpColor(color(255, 255, 255, 100), color(0, 255, 100), isAligned ? 1 : map(d, 0, 200, 1, 0));

        stroke(ringColor);

        ellipse(targetX, targetY, 120);

        

        noStroke();

        fill(255);

        textSize(26);

        if (isAligned) {

          fill(0, 255, 100);

          text("TAP TO START", width/2, height/2 + 100);

        } else {

          text("ALIGN NOSE TO START", width/2, height/2 + 100);

        }

      } 

      

      // --- STATE: PLAYING ---

      else if (gameState === 'PLAYING') {

        // --- UPDATED PHYSICS: Slower and floatier ---

        ball.vy += 0.15; // Much lower gravity for a realistic, slower arc

        ball.x += ball.vx;

        ball.y += ball.vy;


        // Wall collisions

        if (ball.x < ball.radius) { ball.x = ball.radius; ball.vx *= -0.8; }

        if (ball.x > width - ball.radius) { ball.x = width - ball.radius; ball.vx *= -0.8; }


        let d = dist(ball.x, ball.y, nose.x, nose.y);

        if (d < ball.radius + nose.radius && ball.vy > 0) {

          ball.vy = -7; // Softer upward bounce velocity

          ball.vx = map(ball.x - nose.x, -nose.radius, nose.radius, -4, 4); // Softer horizontal spin

          score++;

        }


        // Draw Badge

        drawRocketBadge(ball.x, ball.y, ball.radius);


        // Score display

        noStroke();

        fill(255, 180); 

        textSize(64);

        text(score, width/2, 80);


        if (ball.y > height + ball.radius * 2) {

          gameState = 'GAMEOVER';

        }

      } 

      

      // --- STATE: GAMEOVER ---

      else if (gameState === 'GAMEOVER') {

        noStroke();

        fill(255, 50, 50);

        textSize(44);

        text("GAME OVER", width/2, height/2 - 50);

        

        fill(255);

        textSize(30);

        text("SCORE: " + score, width/2, height/2);

        

        fill(200);

        textSize(22);

        text("TAP TO RESTART", width/2, height/2 + 60);

      }

    }


    function drawRocketBadge(x, y, radius) {

      push();

      translate(x, y);

      

      if(ball.vy > 1) {

          ballParticles.push({x: random(-radius/2, radius/2), y: radius/2, a: 150});

      }

      for(let i = ballParticles.length - 1; i >= 0; i--) {

          let p = ballParticles[i];

          fill(255, 150, 100, p.a);

          noStroke();

          ellipse(p.x, p.y + p.a/10, random(5, 10)); 

          p.a -= 10;

          p.y += 1.5; // Smoother particle movement

          if(p.a <= 0) ballParticles.splice(i, 1);

      }

      

      scale(radius/20); 


      // 1. Shield/Badge Background

      noStroke();

      fill(220, 220, 230); 

      beginShape();

      vertex(0, -20);

      vertex(18, -15);

      vertex(20, 10);

      vertex(0, 25);

      vertex(-20, 10);

      vertex(-18, -15);

      endShape(CLOSE);

      fill(180, 180, 190);

      beginShape(); 

      vertex(0, -15);

      vertex(14, -10);

      vertex(15, 8);

      vertex(0, 20);

      vertex(-15, 8);

      vertex(-14, -10);

      endShape(CLOSE);


      // 2. The Rocket Symbol

      // Fins

      fill(50); 

      triangle(-10, 5, -18, 18, -6, 12);

      triangle(10, 5, 18, 18, 6, 12);


      // Body

      fill(200, 50, 50); 

      beginShape();

      vertex(-6, 12);

      vertex(-6, -10);

      bezierVertex(-6,-20, 0, -25, 0, -25); 

      bezierVertex(0,-25, 6, -20, 6, -10);

      vertex(6, 12);

      endShape(CLOSE);


      // Flames/Flickering Fire

      if (gameState === 'PLAYING') {

          fill(255, 150 + random(105), random(100)); 

          let fireY = 12 + random(5, 15);

          ellipse(0, 15, 10, fireY);

          ellipse(-3, 13, 6, fireY/1.5);

          ellipse(3, 13, 6, fireY/1.5);

      }

      

      // Port hole

      fill(100, 200, 255, 150);

      stroke(255, 200);

      strokeWeight(1);

      ellipse(0, -5, 8, 8);

      

      pop();

    }


    function mousePressed() {

      if (gameState === 'BLOCKED') return false;

      if (gameState === 'ALIGN') {

        if (dist(nose.x, nose.y, width/2, height/2) < 60) {

          resetGame();

          gameState = 'PLAYING';

        }

      } else if (gameState === 'GAMEOVER') {

        ballParticles = [];

        gameState = 'ALIGN';

      }

      return false; 

    }


    function resetGame() {

      score = 0;

      ball.x = width / 2;

      ball.y = height / 4; 

      ball.vx = random(-2, 2); // Slower initial horizontal speed

      ball.vy = 0;

      ballParticles = [];

    }
